#### Build from source

```bash

-c compile and not assemble
-Wall detailed message
-O optimization
-o output

-lm libm.so or libm.a
// after cd

```

```bash
LIBS = -lm
OBJS = hello.o 
CFLAGS = -Wall
main: ${OBJS}
	gcc -o $@ $(OBJS) ${LIBS}
clean:
	rm -f main ${OBJS}
```
